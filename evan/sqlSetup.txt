CREATE TABLE accounts(User_ID char(10) PRIMARY KEY NOT NULL,Balance numeric(15,2) NOT NULL);

CREATE TABLE transactions(Trans_Key serial PRIMARY KEY,User_ID char(10) REFERENCES accounts(User_ID),Stock_ID char(3),Adjustment numeric(15,2) NOT NULL,Timestamp bigint NOT NULL,Command char NOT NULL);

CREATE TABLE triggers(User_ID char(10) REFERENCES accounts(User_ID),Stock_ID char(3) NOT NULL,Trigger_Type char NOT NULL,Reserved_Funds numeric(15,2) NOT NULL,Trigger_Value numeric(15,2) NOT NULL);

CREATE TABLE stocks(User_ID char(10) REFERENCES accounts(User_ID),Stock_ID char(3) NOT NULL,Amount integer NOT NULL);
